## 三、数据库设计

### 3.1 作物表（crops）

**集合名**：`crops`

```typescript
interface Crop {
  _id: string;                    // 文档ID（自动生成）
  name: string;                   // 作物名称，如"番茄"
  icon: string;                   // 图标URL或本地路径
  category: string;               // 分类：叶菜类|瓜果类|豆类|根茎类|葱蒜类
  difficulty: string;             // 难度：简单|中等|较难
  growthCycle: number;            // 生长周期（天）
  suitableProvinces: string[];    // 适宜省份，如["广东省", "广西壮族自治区"]
  suitableMonths: {               // 各省份适宜月份
    [province: string]: number[]; // 如 "广东省": [2, 3, 4, 5, 6, 7, 8, 9, 10]
  };
  steps: PlantStep[];             // 种植步骤
  tips: string[];                 // 注意事项
  faqs: FAQ[];                    // 常见问题
  createdAt: Date;                // 创建时间
  updatedAt: Date;                // 更新时间
}

interface PlantStep {
  order: number;                  // 步骤顺序
  title: string;                  // 步骤标题，如"播种"
  description: string;            // 步骤描述
  duration: number;               // 该步骤持续天数
  tips: string[];                 // 该步骤的注意事项
}

interface FAQ {
  question: string;               // 问题
  answer: string;                 // 答案
}
```

## 四、API 设计

### 4.1 获取作物列表

**云函数**：`getCrops`

**请求参数**：
```typescript
interface GetCropsRequest {
  province: string;    // 省份，如"广东省"
  month?: number;      // 月份（1-12），不传则返回该省所有作物
  category?: string;   // 分类筛选，不传则返回所有分类
}
```

**返回数据**：
```typescript
interface GetCropsResponse {
  code: number;        // 0: 成功
  message: string;
  data: {
    crops: Crop[];
    total: number;
  };
}
```

**数据库查询逻辑**：
```javascript
const db = cloud.database();
const _ = db.command;

// 查询该省份且适宜该月份的作物
let query = {
  suitableProvinces: province
};

if (month) {
  query[`suitableMonths.${province}`] = _.in([month]);
}

if (category) {
  query.category = category;
}

const result = await db.collection('crops').where(query).get();
```

  // 1. 获取用户位置
  this.getUserLocation();

```javascript
// 获取用户位置
async getUserLocation() {
  try {
    const res = await wx.getLocation({ type: 'wgs84' });
    // 调用逆地理编码获取省份
    // ...
  } catch (err) {
    // 定位失败，使用默认省份
    console.log('定位失败，使用默认省份');
  }
}

// 加载推荐作物
async loadRecommendedCrops() {
  const { province, currentMonth } = this.data;
  
  wx.cloud.callFunction({
    name: 'getCrops',
    data: { province, month: currentMonth },
    success: res => {
      this.setData({
        recommendedCrops: res.result.data.crops,
        loading: false
      });
    }
  });
}

// 切换省份
onProvinceChange(e) {
  const province = e.detail.value;
  this.setData({ province });
  this.loadRecommendedCrops();
}
```

async loadPlantings() {
  wx.cloud.callFunction({
    name: 'getMyPlantings',
    data: { status: 'growing' },
    success: res => {
      // 计算生长天数
      const plantings = res.result.data.plantings.map(p => ({
        ...p,
        growthDays: Math.floor((new Date() - new Date(p.startDate)) / (1000 * 60 * 60 * 24))
      }));
